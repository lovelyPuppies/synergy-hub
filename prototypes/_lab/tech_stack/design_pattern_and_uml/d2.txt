    Expressions = Reference
      [": ", Label,
        ["{" PropertyDefinition  "}"]
      ];

    - Reference: (Container | Realtionship)
    
    - Container = Entity, {".", Entity};
    - Entity = "_" | CustomEntity
      // "_" means the Parent

    - Realtionship = Container, Connection, Container
    - Connection = "--" | "->" | "<-" | "<->"

    - PropertyDefinition =
      ({Container, "."}, Property, ": ", Value)
      | Expressions;
